4-function Calculator:
$ define variables
var(num1)
var(num2)
var(operator)
var(result)
$ get user input
paste(input"enter first number: " => num1)
paste(input"enter second number: " => num2)
paste(input"enter operator {+|-|*|/}: " => operator)
$ perform calculation
if(@operator == "+") <<
  add(@num1, @num2 => result)
  >>
if(@operator == "-") <<
  sub(@num1, @num2 => result)
  >>
if(@operator == "*") <<
  mul(@num1, @num2 => result)
  >>
if(@operator == "/") <<
  div(@num1, @num2 => result)
  >>
paste(text"result: @result")
;
Fibonacci Sequence:
$ define variables
var(num1 = 0)
var(num2 = 1)
var(temp)
$ generate infinite fibonacci sequence
loop(0) <<
  paste(text"@num1")
  add(@num1, @num2 => temp)
  var(num1 = @num2)
  var(num2 = @temp)
  >>
;
FizzBuzz:
$ define variables
var(temp1)
var(temp2)
$ loop through numbers 1 to 100 (use built-in loop counter as number index)
loop(100, num) <<
  mod(@loop:num, 3 => temp1)
  mod(@loop:num, 5 => temp2)
  if(@temp1 == 0) <<
    if(@temp2 == 0) <<
      paste(text"fizzbuzz")
      >>
    else() <<
      paste(text"fizz")
      >>
    >>
    else() <<
      if(@temp2 == 0) <<
      paste(text"buzz")
      >>
    else() <<
      paste(text"@loop:num")
      >>
    >>
  >>
;
Reverse a String:
$ define variables
var(input)
var(length)
var(reversed)
var(letter)
var(index)
$ get user input
paste(input"enter a string: " => input)
length(@input => length)
var(index = @length)
$ build the reversed string
loop(@length) <<
  letter(@index, @input => letter)
  concat(@reversed, @letter => reversed)
  sub(@index, 1 => index)
  >>
paste(text"reversed string: @reversed")
;
